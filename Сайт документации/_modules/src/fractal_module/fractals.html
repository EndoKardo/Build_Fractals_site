  def build_fractl(self, fractal_name: str, fractal_data: bytes):

  def standart_patterns(self, standart_patterns: dict[str], standart_patterns_data: dict[bytes]):
    """
    Стандартные шаблоны фракталов

    Args:
      standart_patterns (dict): Словарь содержащий имена новых шаблонов (str)
      standart_patterns_data (dict): Словарь содержащий данные о шаблоннах (bytes)

    Returns:
      bool: Сообщает об успешном(True) или неуспешном(False) получении стандартных шаблонов
    """
    try:
      for i in standart_patterns:
        self.patterns[i] = standart_patterns_data
      return True
    except:
      return False
